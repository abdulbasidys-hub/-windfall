rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /windfall_distributions/{doc} {
      allow read: if true;
      allow write: if false;
    }
    match /windfall_stats/{doc} {
      allow read: if true;
      allow write: if false;
    }
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
